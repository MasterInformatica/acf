\begin{frame}
  \frametitle{Lógica epistémica (1 agente)}
  \vspace{0.8cm}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      Lógica preposicional:
      \begin{itemize}
      \item $p, q, \cdots \in \mathcal{P}$
      \item $\varphi,\quad \neg\varphi,\quad \varphi\mor\psi,\quad \varphi\mand\psi$
      \item $\varphi \rightarrow \psi $ ( $\myeq \neg\varphi \mor \psi$ )
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \visible<2->{
        Lógica epistémica:
        \begin{itemize}
        \item $\K\,\varphi$\hspace{2.5cm}
        \item $\M\,\varphi \myeq \neg\K\neg\varphi$\hspace{0.9cm}
        \end{itemize}
      }
    \end{column}
  \end{columns}
  \vspace{1cm}
  \visible<3->{
    Algunos ejemplos:
    \begin{align*}
      \K(\varphi \rightarrow \psi): & $\hspace{0.3cm} "Se sabe que $\varphi$ implica $\psi$"$\\
      \K \varphi \mor \K \neg \varphi: & $\hspace{0.3cm} "Se sabe que
                                         $\varphi$ o $\neg\varphi$"$\\
      \K \varphi \mor \neg \K \varphi:  &$\hspace{0.3cm} "Puede que se sepa
                                          o no que $\varphi$"$\\
    \end{align*}
  }

\end{frame}
